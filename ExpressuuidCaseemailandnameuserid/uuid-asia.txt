uuid.NIL 	        --  The nil UUID string (all zeros)

uuid.validate() 	--  Test a string to see if it is a valid UUID

uuid.v5()	        --  Create a version 5 (namespace w/ SHA-1) UUID 
uuid.stringify() 	--  Convert array of bytes to UUID string
